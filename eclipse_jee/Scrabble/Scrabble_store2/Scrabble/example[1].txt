  {
    if(document.images)
    {
      for(var i=0;i<6;i++)
      {
        var j = 1+i+Math.random()*(6-i);
        j = Math.floor(j);
        var imgi = document["tray"+i].src;
        var imgj = document["tray"+j].src;
        document["tray"+i].src=imgj;
        document["tray"+j].src=imgi;
      }

  function onSortTray()
  {
    if(document.images)
    {
      var s = [];

      s.sort();
